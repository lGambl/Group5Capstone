namespace StudyWeb.Models
{
    /// <summary>
    ///   The RegisterDto class in the Web app.
    /// </summary>
    public class RegisterDto
    {

        ///   Gets or sets the username.
        /// </summary>
        /// <value>The username.</value>

        ///   Gets or sets the password.
        /// </summary>
